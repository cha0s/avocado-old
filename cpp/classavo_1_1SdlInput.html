avo::SdlInput::SdlInput ( )
: Input()
, numJoysticks(0)
, joystick(NULL)
{
numJoysticks = SDL_NumJoysticks();
if (numJoysticks > 0) {
SDL_JoystickEventState(SDL_ENABLE);
joystick = new SDL_Joystick *[numJoysticks];
for (int i = 0; i < numJoysticks; i++) {
joystick[i] = SDL_JoystickOpen(i);
}
}
}
