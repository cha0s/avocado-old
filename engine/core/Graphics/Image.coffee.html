avo.Image.blendPixel = (src, dst, alpha = 255) ->
  return unless src? and dst?
  
  {r, g, b, a} = Rgba src
  sr = r; sg = g; sb = b; sa = a;
  
  return dst unless sa > 0
  
  {r, g, b, a} = Rgba dst
  dr = r; dg = g; db = b; da = a;
  
  pAlpha = sa * (alpha / 255)
  
  dr = (sr * pAlpha + dr * (255 - pAlpha)) / 255
  dg = (sg * pAlpha + dg * (255 - pAlpha)) / 255
  db = (sb * pAlpha + db * (255 - pAlpha)) / 255
  da = pAlpha
  
  Rgba dr, dg, db, da
